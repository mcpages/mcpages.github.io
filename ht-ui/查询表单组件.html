# 查询表单组件

# 基本用法


<ht-search-form
  ref="htSearchForm"
  :queryForm="queryForm"
  @handleQuery="handleFilter"
/>
...data...

// 查询表单
queryForm: [
  {
    itemType: 'text', // 输入框类型
    labelName: '供应商名称', // 输入框名字
    propName: 'name', // 输入框字段名
    placeholder: '供应商名称' // 输入框placeholder提示语加上，可用于告知用户规则
  },
  ...
  {
    itemType: 'dateRange', // 日期范围类型
    labelName: '日期',
    propName: 'date'

  }
],
queryFormSelected: ['院级供应商', '供应商名称2', '社会信用代码', '厂所供应商', '日期'],

....methods....

handleFilter (queryData) {
  console.log('queryData', queryData)
},

# 属性(Attributes)

参数 说明 类型 默认值 可选值
queryForm 查询表单内容 Array [] -
advanced 是否高级查询 Boolean false -
queryFormSelected 展示的表单属性 [] [] -
hiddenLength 多于几个属性可展开 Number 5 -

# 方法(Methods)

方法名 说明 参数
handleQuery 查询按钮触发 queryForm

# 插槽 Slot

name 说明
advance 插入至右侧高级设置，可以根据选中的checkbox显示隐藏哪些属性，详细使用方法可以参照拖拽排序按钮组
