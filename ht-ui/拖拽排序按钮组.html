# 拖拽排序按钮组

# 基本用法

<HtDragCheckbox
  :allList="allList"
  :checkList="checkList"
  :labelName="'labelName'"
  @changeSelected="changeSelected"
></HtDragCheckbox>

...data
  allList:[],
  checkList:[]

...methods
changeSelected (values) {
  this.checkList = values
},

# 高级用法 支持拖拽排序

<HtDragCheckbox
  :allList="allList"
  :checkList="checkList"
  :labelName="'labelName'"
  :draggable="true"
  @changeAllList="changeAllList"
  @changeSelected="changeSelected"
></HtDragCheckbox>

...data
  allList:[],
  checkList:[]

...methods

changeAllList (values) {
  this.allList = values
},
changeSelected (values) {
  this.checkList = values
},

# 属性(Attributes)

参数 说明 类型 默认值 可选值
labelName 标签展示 String 'label' -
allList 所有checkbox数据 Array [] -
checkList 选中的数组 Array [] -

# 方法(Methods)

方法名 说明 参数
changeAllList 拖拽后的排序结果 values
changeSelected 点击选中的结果 values
