# 树形穿梭框组件


<tree-transfer
  ref="treeTransfer"
  :data-source.sync="dataSource"
  :defaultCheckedKeys="defaultValue"
  @left-check-change="handleLeftCheckChange"
  @right-check-change="handleRightCheckChange"
  @change="handleChange"
  :isRadio="false"
  :filterable="true"
  :father-choose="false"
  >
</tree-transfer>

....data

dataSource: [],
defaultValue: []


...methods

handleChange (value, direction, currentKeys) {
  console.log('handleChange', value, direction, currentKeys)
},
handleLeftCheckChange (value, currentKeys) {
  console.log('handleLeftCheckChange:', value, currentKeys)
},
handleRightCheckChange (value, currentKeys) {
  console.log('handleRightCheckChange:', value, currentKeys)
}

# 属性(Attributes)

参数 说明 类型 默认值 可选值
width 宽度 String '800px' -
height 高度 String '500px' -
titles 标题 Array ['源数据', '目标源数据'] -
dataSource 源数据 Array [] -
defaultProps el-tree 配置项 Object { label: 'label', children: 'children' } -
nodeKey el-tree node-key 必须唯一 String 'id' -
filterable 是否启用筛选 Boolean false -
openAll 是否展开所有节点 Boolean true -
defaultCheckedKeys 默认选中节点 Array [] -
defaultExpandedKeys 默认展开节点 Array [] -
filterPlaceholder 筛选placeholder Array '请输入搜索内容' -
accordion 是否每次只打开一个同级树节点 Boolean false -
renderAfterExpand 是否在第一次展开某个树节点后才渲染其子节点 Boolean true -
expandOnClickNode 是否在点击节点的时候展开或者收缩节点 Boolean true -
fatherChoose 是否严格的遵循父子不互相关联的做法,父节点是否可选 Boolean false -
isRadio 是否单选 Boolean false -
buttonTexts 穿梭按钮文案 Array [] -

# 方法(Methods)

方法名 说明 参数
handleChange 穿梭事件 value, direction, currentKeys
handleLeftCheckChange 左侧checkbox选中事件 value, currentKeys
handleRightCheckChange 右侧checkbox选中事件 value, currentKeys
